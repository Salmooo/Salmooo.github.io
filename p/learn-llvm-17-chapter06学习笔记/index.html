2. 抛出和捕获异常

2.1 C++ 异常处理机制概述

LLVM IR 的异常处理与平台（libunwind）紧密相关。C++ 运行时提供以下关键函数：

函数 作用
__cxa_allocate_exception(size) 为异常分配内存
__cxa_throw(ptr, typeinfo, destructor) 抛出异常，启动栈展开，永不返回
__cxa_begin_catch(ptr) 开始处理异常，返回有效负载指针
__cxa_end_catch() 结束处理，释放异常内存
__gxx_personality_v0 个性函数，参与栈展开决策

抛出 int 的 IR 示例：

1
2
3
4

%eh = call ptr @__cxa_allocate_exception(i64 4)
store i32 1, ptr %eh
call void @__cxa_throw(ptr %eh, ptr @_ZTIi, ptr null)
unreachable

_ZTIi 是 C++ 的 int 类型信息（typeinfo）。

2.2 invoke 与 landingpad

对可能抛异常的调用，必须用 invoke 而非 call：

1
2

%y = invoke i32 @_Z3bari(i32 %x) to label %next
         unwind label %lpad
  • 正常返回：跳转到 %next
  • 异常：跳转到 %lpad（着陆垫）

着陆垫 (landingpad) 必须以 landingpad 指令开始：

1
2
3
4
5

lpad:
  %exc = landingpad { ptr, i32 }
          cleanup
          catch ptr @_ZTIi
          filter [1 x ptr] [ptr @_ZTIi]
子句类型 含义
cleanup 存在清理代码，栈展开时调用
catch ptr T 可处理的异常类型
filter [N x ptr] [...] 不在列表中的类型会触发（用于 throw 规范）

landingpad 返回值：{ ptr, i32 }，第一个是指向异常的指针，第二个是类型选择器。

  • 类型选择器 > 0：与 catch 匹配
  • 类型选择器 < 0：与 filter 不匹配
  • 类型选择器 = 0：需执行 cleanup

void createICmpEq(Value *Left, Value *Right,
                  BasicBlock *&TrueDest, BasicBlock *&FalseDest,
                  const Twine &TrueLabel = "", const Twine &FalseLabel = "") {
  Function *Fn = Builder.GetInsertBlock()->getParent();
  TrueDest = BasicBlock::Create(M->getContext(), TrueLabel, Fn);
  FalseDest = BasicBlock::Create(M->getContext(), FalseLabel, Fn);
  Value *Cmp = Builder.CreateCmp(CmpInst::ICMP_EQ, Left, Right);
  Builder.CreateCondBr(Cmp, TrueDest, FalseDest);
}

用于比较两个值并创建分支到两个新基本块。

void addThrow(int PayloadVal) {
  if (!TypeInfo) {
    // 1. 创建 int 类型信息全局变量
    TypeInfo = new GlobalVariable(*M, Int8PtrTy,
                                   /*isConstant=*/true,
                                   GlobalValue::ExternalLinkage,
                                   /*Initializer=*/nullptr, "_ZTIi");

    // 2. 声明 __cxa_allocate_exception 和 __cxa_throw
    createFunc(AllocEHFty, AllocEHFn, "__cxa_allocate_exception",
               Int8PtrTy, {Int64Ty});
    createFunc(ThrowEHFty, ThrowEHFn, "__cxa_throw", VoidTy,
               {Int8PtrTy, Int8PtrTy, Int8PtrTy});

    // 3. 声明个性函数并绑定到当前函数
    FunctionType *PersFty;
    Function *PersFn;
    createFunc(PersFty, PersFn, "__gxx_personality_v0",
               Int32Ty, std::nullopt, true);
    Function *Fn = Builder.GetInsertBlock()->getParent();
    Fn->setPersonalityFn(PersFn);

    // 4. 创建着陆垫基本块并填充 addLandingPad()
    BasicBlock *SaveBB = Builder.GetInsertBlock();
    LPadBB = BasicBlock::Create(M->getContext(), "lpad", Fn);
    Builder.SetInsertPoint(LPadBB);
    addLandingPad();

    // 5. 创建 unreachable 基本块
    UnreachableBB = BasicBlock::Create(M->getContext(), "unreachable", Fn);
    Builder.SetInsertPoint(UnreachableBB);
    Builder.CreateUnreachable();
    Builder.SetInsertPoint(SaveBB);
  }

  // 6. 分配异常内存（4 字节 int）
  Constant *PayloadSz = ConstantInt::get(Int64Ty, 4, false);
  CallInst *EH = Builder.CreateCall(AllocEHFty, AllocEHFn, {PayloadSz});

  // 7. 存储有效负载
  Value *PayloadPtr = Builder.CreateBitCast(EH, Int32PtrTy);
  Builder.CreateStore(ConstantInt::get(Int32Ty, PayloadVal, true), PayloadPtr);

  // 8. 使用 invoke 调用 __cxa_throw（正常→UnreachableBB，异常→LPadBB）
  Builder.CreateInvoke(ThrowEHFty, ThrowEHFn, UnreachableBB, LPadBB,
                       {EH, ConstantExpr::getBitCast(TypeInfo, Int8PtrTy),
                        ConstantPointerNull::get(Int8PtrTy)});
}

void addLandingPad() {
  // 声明 llvm.eh.typeid.for, __cxa_begin_catch, __cxa_end_catch, puts
  createFunc(TypeIdFty, TypeIdFn, "llvm.eh.typeid.for", Int32Ty, {Int8PtrTy});
  createFunc(BeginCatchFty, BeginCatchFn, "__cxa_begin_catch", Int8PtrTy, {Int8PtrTy});
  createFunc(EndCatchFty, EndCatchFn, "__cxa_end_catch", VoidTy);
  createFunc(PutsFty, PutsFn, "puts", Int32Ty, {Int8PtrTy});

  // 1. 创建 landingpad 指令，catch int 类型
  LandingPadInst *Exc = Builder.CreateLandingPad(
      StructType::get(Int8PtrTy, Int32Ty), 1, "exc");
  Exc->addClause(ConstantExpr::getBitCast(TypeInfo, Int8PtrTy));

  // 2. 提取类型选择器，获取 int 的 typeid
  Value *Sel = Builder.CreateExtractValue(Exc, {1}, "exc.sel");
  CallInst *Id = Builder.CreateCall(TypeIdFty, TypeIdFn,
                                   {ConstantExpr::getBitCast(TypeInfo, Int8PtrTy)});

  // 3. 比较：匹配则进入 match，否则 resume
  BasicBlock *TrueDest, *FalseDest;
  createICmpEq(Sel, Id, TrueDest, FalseDest, "match", "resume");

  Builder.SetInsertPoint(FalseDest);
  Builder.CreateResume(Exc);   // 不匹配，交回运行时

  Builder.SetInsertPoint(TrueDest);
  Value *Ptr = Builder.CreateExtractValue(Exc, {0}, "exc.ptr");
  Builder.CreateCall(BeginCatchFty, BeginCatchFn, {Ptr});

  Value *MsgPtr = Builder.CreateGlobalStringPtr("Divide by zero!", "msg", 0, M);
  Builder.CreateCall(PutsFty, PutsFn, {MsgPtr});
  Builder.CreateCall(EndCatchFty, EndCatchFn);
  Builder.CreateRet(Int32Zero);
}

3. 基于类型的别名分析（TBAA）元数据

3.1 问题背景 

1
2
3
4

void doSomething(int *p, float *q) {
  *p = 42;
  *q = 3.1425;
}

优化器无法判断 p 和 q 是否指向同一内存。若可证明不别名，则可重排存储顺序等优化。

3.2 TBAA 类型层次结构

  • 根节点：类型层次结构的起点
  • 标量类型：INTEGER、char、any pointer 等
  • 聚合类型：结构体，引用标量类型和偏移量

访问标签：(基类型, 访问类型, 偏移量)

  • 标量：访问类型=基类型，偏移=0
  • 聚合：描述某偏移处成员访问

别名判定：若两个访问的 (基类型, 偏移量) 在父关系中相关，则可能别名。

// 根节点：惰性创建
llvm::MDNode *CGTBAA::getRoot() {
  if (!Root)
    Root = MDHelper.createTBAARoot("Simple tinylang TBAA");
  return Root;
}

// 标量类型节点
llvm::MDNode *CGTBAA::createScalarTypeNode(TypeDeclaration *Ty,
                                           StringRef Name,
                                           llvm::MDNode *Parent) {
  llvm::MDNode *N = MDHelper.createTBAAScalarTypeNode(Name, Parent);
  return MetadataCache[Ty] = N;
}

// 结构体类型节点
llvm::MDNode *CGTBAA::createStructTypeNode(
    TypeDeclaration *Ty, StringRef Name,
    llvm::ArrayRef<std::pair<llvm::MDNode *, uint64_t>> Fields) {
  llvm::MDNode *N = MDHelper.createTBAAStructTypeNode(Name, Fields);
  return MetadataCache[Ty] = N;
}

// 根据 tinylang 类型生成 TBAA 元数据
llvm::MDNode *CGTBAA::getTypeInfo(TypeDeclaration *Ty) {
  if (llvm::MDNode *N = MetadataCache[Ty])
    return N;

  if (auto *Pervasive = llvm::dyn_cast<PervasiveTypeDeclaration>(Ty)) {
    return createScalarTypeNode(Pervasive, Pervasive->getName(), getRoot());
  }
  if (auto *Pointer = llvm::dyn_cast<PointerTypeDeclaration>(Ty)) {
    return createScalarTypeNode(Pointer, "any pointer", getRoot());
  }
  if (auto *Array = llvm::dyn_cast<ArrayTypeDeclaration>(Ty)) {
    return createScalarTypeNode(Array, Array->getType()->getName(), getRoot());
  }
  if (auto *Record = llvm::dyn_cast<RecordTypeDeclaration>(Ty)) {
    llvm::SmallVector<std::pair<llvm::MDNode *, uint64_t>, 4> Fields;
    auto *Rec = llvm::cast<llvm::StructType>(CGM.convertType(Record));
    const llvm::StructLayout *Layout =
        CGM.getModule()->getDataLayout().getStructLayout(Rec);
    unsigned Idx = 0;
    for (const auto &F : Record->getFields()) {
      uint64_t Offset = Layout->getElementOffset(Idx);
      Fields.emplace_back(getTypeInfo(F.getType()), Offset);
      ++Idx;
    }
    return createStructTypeNode(Record, CGM.mangleName(Record), Fields);
  }
  return nullptr;
}

llvm::MDNode *CGTBAA::getAccessTagInfo(TypeDeclaration *Ty) {
  return getTypeInfo(Ty);
}

// 基本类型
llvm::DIType *CGDebugInfo::getPervasiveType(TypeDeclaration *Ty) {
  if (Ty->getName() == "INTEGER")
    return DBuilder.createBasicType(Ty->getName(), 64, llvm::dwarf::DW_ATE_signed);
  if (Ty->getName() == "BOOLEAN")
    return DBuilder.createBasicType(Ty->getName(), 1, llvm::dwarf::DW_ATE_boolean);
  llvm::report_fatal_error("Unsupported pervasive type");
}

// 类型别名
llvm::DIType *CGDebugInfo::getAliasType(AliasTypeDeclaration *Ty) {
  return DBuilder.createTypedef(getType(Ty->getType()), Ty->getName(),
                                CU->getFile(), getLineNumber(Ty->getLocation()),
                                getScope());
}

llvm::DIType *CGDebugInfo::getArrayType(ArrayTypeDeclaration *Ty) {
  auto *ATy = llvm::cast<llvm::ArrayType>(CGM.convertType(Ty));
  const llvm::DataLayout &DL = CGM.getModule()->getDataLayout();
  uint64_t NumElements = ...;  // 从 Nums 表达式获取
  llvm::SmallVector<llvm::Metadata *, 4> Subscripts;
  Subscripts.push_back(DBuilder.getOrCreateSubrange(0, NumElements));
  return DBuilder.createArrayType(
      DL.getTypeSizeInBits(ATy) * 8,
      1 << Log2(DL.getABITypeAlign(ATy)),  // 对齐（字节）
      getType(Ty->getType()),
      DBuilder.getOrCreateArray(Subscripts));
}

void CGDebugInfo::emitGlobalVariable(VariableDeclaration *Decl,
                                      llvm::GlobalVariable *V) {
  llvm::DIGlobalVariableExpression *GV = DBuilder.createGlobalVariableExpression(
      getScope(), Decl->getName(), V->getName(),
      CU->getFile(), getLineNumber(Decl->getLocation()),
      getType(Decl->getType()), false);
  V->addDebugInfo(GV);
}

void CGDebugInfo::emitProcedure(ProcedureDeclaration *Decl, llvm::Function *Fn) {
  llvm::DISubroutineType *SubT = getType(Decl);
  llvm::DISubprogram *Sub = DBuilder.createFunction(
      getScope(), Decl->getName(), Fn->getName(),
      CU->getFile(), getLineNumber(Decl->getLocation()),
      SubT, getLineNumber(Decl->getLocation()),
      llvm::DINode::FlagPrototyped,
      llvm::DISubprogram::SPFlagDefinition);
  openScope(Sub);
  Fn->setSubprogram(Sub);
}

void CGDebugInfo::emitProcedureEnd(ProcedureDeclaration *Decl, llvm::Function *Fn) {
  if (Fn && Fn->getSubprogram())
    DBuilder.finalizeSubprogram(Fn->getSubprogram());
  closeScope();
}

llvm::DILocalVariable *CGDebugInfo::emitParameterVariable(
    FormalParameterDeclaration *FP, size_t Idx,
    llvm::Value *Val, llvm::BasicBlock *BB) {
  llvm::DILocalVariable *Var = DBuilder.createParameterVariable(
      getScope(), FP->getName(), Idx, CU->getFile(),
      getLineNumber(FP->getLocation()), getType(FP->getType()));
  DBuilder.insertDbgValueIntrinsic(Val, Var, DBuilder.createExpression(),
                                   getDebugLoc(FP->getLocation()), BB);
  return Var;
}

void CGDebugInfo::emitValue(llvm::Value *Val, llvm::DILocalVariable *Var,
                            SMLoc Loc, llvm::BasicBlock *BB) {
  llvm::DebugLoc DLoc = getDebugLoc(Loc);
  llvm::Instruction *Instr = DBuilder.insertDbgValueIntrinsic(
      Val, Var, DBuilder.createExpression(), DLoc, BB);
  Instr->setDebugLoc(DLoc);
}

llvm::DebugLoc CGDebugInfo::getDebugLoc(SMLoc Loc) {
  std::pair<unsigned, unsigned> LineAndCol =
      CGM.getASTCtx().getSourceMgr().getLineAndColumn(Loc);
  llvm::DILocation *DILoc = llvm::DILocation::get(
      CGM.getLLVMCtx(), LineAndCol.first, LineAndCol.second, getScope());
  return llvm::DebugLoc(DILoc);
}

// CGModule.cpp
void CGModule::applyLocation(llvm::Instruction *Inst, llvm::SMLoc Loc) {
  if (CGDebugInfo *Dbg = getDbgInfo())
    Inst->setDebugLoc(Dbg->getDebugLoc(Loc));
}

5. 总结

特性 关键 API / 指令 作用
异常抛出 __cxa_allocate_exception, __cxa_throw, invoke 分配异常、抛出、双路控制流
异常捕获 landingpad, llvm.eh.typeid.for, __cxa_begin_catch, resume 类型匹配、处理、恢复
TBAA MDBuilder::createTBAARoot, createTBAAScalarTypeNode, createTBAAStructTypeNode 为 load/store 添加 !tbaa 元数据
调试信息 DIBuilder, createCompileUnit, createFunction, insertDbgValueIntrinsic 支持 gdb/lldb 源码调试

本章为编译器增加了异常处理、TBAA 优化支持和调试信息生成能力，是生产级编译器的重要进阶特性。
